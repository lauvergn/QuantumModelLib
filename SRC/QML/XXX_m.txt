!> @brief Module which makes the initialization, calculation of the XXX potentials (value, gradient and hessian).
!!
!> @author David Lauvergnat
!! @date 07/01/2020
!!
MODULE QML_XXX_m
  USE mod_NumParameters
  USE QML_Empty_m
  IMPLICIT NONE

  PRIVATE

!> @brief Derived type in which the XXX parameters are set-up.
  TYPE, EXTENDS (QML_Empty_t) ::  XXX_Model_t

   PRIVATE

   CONTAINS
    PROCEDURE :: Eval_QModel_Pot => eval_XXX_Pot
    PROCEDURE :: Write_QModel    => Write_XXX_Model
    PROCEDURE :: Write0_QModel   => Write0_XXX_Model
  END TYPE XXX_Model_t

  PUBLIC :: XXX_Model_t,Init_XXX_Model

  CONTAINS
!> @brief Function which makes the initialization of the XXX parameters.
!!
!! @param QModel             TYPE(XXX_Model_t):   result derived type in which the parameters are set-up.
!! @param QModel_in          TYPE(QML_Empty_t):  type to transfer ndim, nsurf ...
!! @param nio_param_file     integer:             file unit to read the parameters.
!! @param read_param         logical:             when it is .TRUE., the parameters are read. Otherwise, they are initialized.
  FUNCTION Init_XXX_Model(QModel_in,read_param,nio_param_file) RESULT(QModel)
  IMPLICIT NONE

    TYPE (XXX_Model_t)                           :: QModel ! RESULT

    TYPE(QML_Empty_t),          intent(in)      :: QModel_in ! variable to transfer info to the init
    integer,                     intent(in)      :: nio_param_file
    logical,                     intent(in)      :: read_param


    !----- for debuging --------------------------------------------------
    character (len=*), parameter :: name_sub='Init_XXX_Model'
    !logical, parameter :: debug = .FALSE.
    logical, parameter :: debug = .TRUE.
    !-----------------------------------------------------------
    IF (debug) THEN
      write(out_unitp,*) 'BEGINNING ',name_sub
      flush(out_unitp)
    END IF

    CALL Init0_QML_Empty(QModel%QML_Empty_t,QModel_in)

    QModel%nsurf    = x
    QModel%ndim     = x
    QModel%pot_name = 'xxx'


    IF (debug) write(out_unitp,*) 'init Q0 of XXX'
    QModel%Q0 = ....

    IF (debug) write(out_unitp,*) 'init d0GGdef of XXX'
    CALL Init_IdMat(QModel%d0GGdef,QModel%ndim)


    IF (debug) THEN
      write(out_unitp,*) 'QModel%pot_name: ',QModel%pot_name
      write(out_unitp,*) 'END ',name_sub
      flush(out_unitp)
    END IF

  END FUNCTION Init_XXX_Model
!> @brief Subroutine wich prints the current XXX_Model parameters.
!!
!! @param QModel            CLASS(XXX_Model_t):   derived type in which the parameters are set-up.
!! @param nio               integer:              file unit to print the parameters.
  SUBROUTINE Write_XXX_Model(QModel,nio)
  IMPLICIT NONE

    CLASS(XXX_Model_t),   intent(in) :: QModel
    integer,              intent(in) :: nio

  END SUBROUTINE Write_XXX_Model
!> @brief Subroutine wich prints the default XXX_Model parameters.
!!
!! @param QModel            CLASS(XXX_Model_t):   derived type in which the parameters are set-up.
!! @param nio               integer:              file unit to print the parameters.
  SUBROUTINE Write0_XXX_Model(QModel,nio)
  IMPLICIT NONE

    CLASS(XXX_Model_t),   intent(in) :: QModel
    integer,              intent(in) :: nio

    write(nio,*) 'XXX default parameters'
    write(nio,*)
    write(nio,*)
    write(nio,*) 'end XXX default parameters'


  END SUBROUTINE Write0_XXX_Model

!> @brief Subroutine wich calculates the XXX potential with derivatives up to the 2d order.
!!
!! @param QModel             CLASS(XXX_Model_t):   derived type in which the parameters are set-up.
!! @param Mat_OF_PotDia(:,:) TYPE (dnS_t):         derived type with the potential (pot),  the gradient (grad) and the hessian (hess).
!! @param dnQ(:)             TYPE (dnS_t)          value for which the potential is calculated
!! @param nderiv             integer:              it enables to specify up to which derivatives the potential is calculated:
!!                                                 the pot (nderiv=0) or pot+grad (nderiv=1) or pot+grad+hess (nderiv=2).
  SUBROUTINE eval_XXX_Pot(QModel,Mat_OF_PotDia,dnQ,nderiv)
  USE QML_dnS_m
  IMPLICIT NONE

    CLASS(XXX_Model_t),   intent(in)    :: QModel
    TYPE (dnS_t),         intent(inout) :: Mat_OF_PotDia(:,:)
    TYPE (dnS_t),         intent(in)    :: dnQ(:)
    integer,              intent(in)    :: nderiv


  END SUBROUTINE eval_XXX_Pot

END MODULE QML_XXX_m
